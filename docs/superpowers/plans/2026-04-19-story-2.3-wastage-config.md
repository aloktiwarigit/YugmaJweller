# Story 2.3 — Wastage Configuration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `GET + PATCH /api/v1/settings/wastage` endpoints and a mobile `WastageRow` component so shopkeepers can configure per-category wastage percentages.

**Architecture:** Shared Zod schema in `packages/shared` owns `WastageConfig` and `WASTAGE_DEFAULTS`. `SettingsCache` grows per-key wastage methods (`shop:{id}:settings:wastage`, TTL 60s). Service merges a single-item PATCH over defaults → upserts `wastage_json` JSONB (sparse map `{ "BRIDAL": "2.50" }`). Percent >30% is a domain rule checked in the service layer (422). No migration — `wastage_json` column exists from migration 0006.

**Prerequisites:** This plan assumes Story 2.2 (making-charges) has been merged so `packages/shared/src/schemas/making-charges.schema.ts` and `ProductCategory` exist. If not, define `ProductCategory` locally in `wastage.schema.ts` and deduplicate when 2.2 merges.

**Tech Stack:** TypeScript strict, Zod 3, NestJS 10, PostgreSQL 15 (withTenantTx), Vitest, @testcontainers/postgresql, React Native (Expo), NativeWind, @testing-library/react.

  async getWastage(): Promise<WastageConfig[] | null> {
    const key = this.wastageKey();
    try {
      const raw = await this.redis.get(key);
      if (!raw) return null;
      const parsed: unknown = JSON.parse(raw);
      const result = WastageArraySchema.safeParse(parsed);
      if (!result.success) {
        await this.redis.del(key);
        return null;
      }
      return result.data;
    } catch {
      try { await this.redis.del(key); } catch { /* ignore */ }
      return null;
    }
  }

  async setWastage(data: WastageConfig[]): Promise<void> {
    await this.redis.set(this.wastageKey(), JSON.stringify(data), 'EX', this.ttlSec);
  }

  async invalidateWastage(): Promise<void> {
    await this.redis.del(this.wastageKey());
  }

  private wastageKey(): string {
    return `shop:${tenantContext.requireCurrent().shopId}:settings:wastage`;
  }
```

  async upsertWastage(
    category: string,
    percent: string,
  ): Promise<UpdateWastageResult> {
    return withTenantTx(this.pool, async (tx) => {
      const shopId = tenantContext.requireCurrent().shopId;

      // Ensure row exists to allow SELECT FOR UPDATE.
      await tx.query(
        `INSERT INTO shop_settings (shop_id) VALUES ($1) ON CONFLICT (shop_id) DO NOTHING`,
        [shopId],
      );

      const beforeRow = await tx.query<{ wastage_json: Record<string, string> | null }>(
        `SELECT wastage_json FROM shop_settings WHERE shop_id = $1 FOR UPDATE`,
        [shopId],
      );
      const storedMap: Record<string, string> = beforeRow.rows[0]?.wastage_json ?? {};
      const before = Object.keys(storedMap).length === 0
        ? null
        : WASTAGE_DEFAULTS.map((d) => ({ category: d.category, percent: storedMap[d.category] ?? d.percent }));

      const mergedMap = { ...storedMap, [category]: percent };

      const r = await tx.query<{ wastage_json: Record<string, string> }>(
        `INSERT INTO shop_settings (shop_id, wastage_json)
         VALUES ($1, $2::jsonb)
         ON CONFLICT (shop_id)
         DO UPDATE SET wastage_json = $2::jsonb, updated_at = now()
         RETURNING wastage_json`,
        [shopId, JSON.stringify(mergedMap)],
      );
      const afterMap = r.rows[0].wastage_json;
      const after = WASTAGE_DEFAULTS.map((d) => ({
        category: d.category,
        percent: afterMap[d.category] ?? d.percent,
      }));

      return { before, after };
    });
  }
```

    const parsed = PatchWastageSchema.safeParse(dto);

      const errors = parsed.error.issues.map((i) => ({ field: i.path.join('.'), code: i.message }));

      parsed.data.category,
      parsed.data.percent,

    void this.auditWastageUpdate(before, after).catch(() => undefined);

import type { WastageConfig } from '@goldsmith/shared';

interface Props {
  config: WastageConfig;

}

const DECIMAL_RE = /^\d+(\.\d+)?$/;

function validate(value: string): string | null {
  if (!DECIMAL_RE.test(value) || parseFloat(value) <= 0) {
    return t('settings.wastage.errors.VALUE_POSITIVE_REQUIRED');
  }
  if (!DECIMAL_RE.test(value)) {
    return t('settings.wastage.errors.VALUE_FORMAT_INVALID');
  }
  if (parseFloat(value) > 30) {
    return t('settings.wastage.errors.wastage_high');
  }
  return null;
}

export function WastageRow({ config, onChange }: Props): React.ReactElement {
  const [error, setError] = useState<string | null>(null);
  const [inputValue, setInputValue] = useState(config.percent);
  const label = t(`settings.making_charges.categories.${config.category}`);

  useEffect(() => {
    setInputValue(config.percent);
  }, [config.percent]);

  function handleValueChange(v: string): void {
    setInputValue(v);
    onChange({ ...config, percent: v });
  }

  function handleBlur(): void {
    setError(validate(inputValue));
  }

  return (

      <Text style={styles.label}>{label}</Text>

        testID="percent-input"
        value={inputValue}
        onChangeText={handleValueChange}
        onBlur={handleBlur}
        keyboardType="decimal-pad"
        style={styles.input}
      />
      {error !== null && <Text style={styles.error}>{error}</Text>}
    </View>
  );
}

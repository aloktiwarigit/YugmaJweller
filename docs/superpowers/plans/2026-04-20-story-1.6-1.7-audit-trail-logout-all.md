```sql
-- 0009_audit_immutability.sql
-- Layer 1: revoke mutation privileges from the app role.
-- app_user can INSERT (needed by auditLog()) but not UPDATE or DELETE.
REVOKE UPDATE, DELETE ON audit_events FROM app_user;

-- Layer 2: belt-and-suspenders trigger fires for ALL roles, including
-- postgres superuser. SECURITY DEFINER runs as the trigger creator,
-- not the invoking role, so it fires regardless of privilege level.
CREATE OR REPLACE FUNCTION prevent_audit_mutation()
  RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
  BEGIN
    RAISE EXCEPTION 'audit_events is append-only — write via auditLog() only'
      USING ERRCODE = 'restrict_violation';
  END;
$$;

CREATE TRIGGER audit_events_immutable
  BEFORE UPDATE OR DELETE ON audit_events
  FOR EACH ROW EXECUTE FUNCTION prevent_audit_mutation();
```

```typescript
import { Inject, Injectable } from '@nestjs/common';
import type { Pool } from 'pg';
import { withTenantTx } from '@goldsmith/db';

export interface AuditEventRow {
  id: string;
  action: string;
  actorName: string;
  actorRole: string;
  createdAt: string;
  metadata?: Record<string, unknown> | null;
}

export interface AuditLogFilters {
  page: number;
  pageSize: number;
  dateRange?: string;
  category?: string;
}

export interface PaginatedAuditLog {
  events: AuditEventRow[];
  total: number;
}

const CATEGORY_ACTIONS: Record<string, string[]> = {
  login: [
    'AUTH_VERIFY_SUCCESS', 'AUTH_VERIFY_FAILURE', 'AUTH_VERIFY_LOCKED',
    'AUTH_VERIFY_REJECTED', 'AUTH_USER_PROVISIONED', 'AUTH_LOGOUT_ALL',
  ],
  staff: ['STAFF_INVITED', 'STAFF_REVOKED', 'STAFF_ACTIVATED'],
  permissions: ['PERMISSIONS_UPDATED', 'ACCESS_DENIED'],
  settings: [
    'SETTINGS_PROFILE_UPDATED', 'SETTINGS_MAKING_CHARGES_UPDATED',
    'SETTINGS_WASTAGE_UPDATED', 'SETTINGS_RATE_LOCK_UPDATED',
    'SETTINGS_LOYALTY_UPDATED', 'SETTINGS_TRY_AT_HOME_UPDATED',
    'SETTINGS_CUSTOM_ORDER_POLICY_UPDATED', 'SETTINGS_RETURN_POLICY_UPDATED',
    'SETTINGS_NOTIFICATION_PREFS_UPDATED',
  ],
};

function dateFromParam(dateRange: string | undefined): Date {
  const IST_OFFSET_MS = 5.5 * 60 * 60 * 1000;
  const nowUtcMs = Date.now();
  if (dateRange === 'today') {
    const nowIST = new Date(nowUtcMs + IST_OFFSET_MS);
    const istMidnightMs =
      Date.UTC(nowIST.getUTCFullYear(), nowIST.getUTCMonth(), nowIST.getUTCDate()) - IST_OFFSET_MS;
    return new Date(istMidnightMs);
  }
  if (dateRange === '7d') return new Date(nowUtcMs - 7 * 24 * 60 * 60 * 1000);
  return new Date(nowUtcMs - 30 * 24 * 60 * 60 * 1000);
}

@Injectable()
export class AuditLogRepository {
  constructor(@Inject('PG_POOL') private readonly pool: Pool) {}

  async findPaginated(filters: AuditLogFilters): Promise<PaginatedAuditLog> {
    const pageSize = Math.min(filters.pageSize, 50);
    const offset = (filters.page - 1) * pageSize;
    const dateFrom = dateFromParam(filters.dateRange);
    const actions = filters.category ? (CATEGORY_ACTIONS[filters.category] ?? null) : null;

    return withTenantTx(this.pool, async (tx) => {
      const rows = await tx.query<{
        id: string;
        action: string;
        created_at: Date;
        metadata: Record<string, unknown> | null;
        actor_name: string | null;
        actor_role: string | null;
      }>(
        `SELECT
           ae.id,
           ae.action,
           ae.created_at,
           ae.metadata,
           COALESCE(su.display_name, NULL) AS actor_name,
           COALESCE(su.role, NULL)         AS actor_role
         FROM audit_events ae
         LEFT JOIN shop_users su ON su.id = ae.actor_user_id
         WHERE ae.created_at >= $1
           AND ($2::text[] IS NULL OR ae.action = ANY($2))
         ORDER BY ae.created_at DESC
         LIMIT $3 OFFSET $4`,
        [dateFrom, actions, pageSize, offset],
      );

      const countResult = await tx.query<{ total: string }>(
        `SELECT COUNT(*)::text AS total
         FROM audit_events ae
         WHERE ae.created_at >= $1
           AND ($2::text[] IS NULL OR ae.action = ANY($2))`,
        [dateFrom, actions],
      );

      const total = parseInt(countResult.rows[0]?.total ?? '0', 10);

      const events: AuditEventRow[] = rows.rows.map((r) => ({
        id: r.id,
        action: r.action,
        actorName: r.actor_name ?? 'System',
        actorRole: r.actor_role ?? 'system',
        createdAt: r.created_at.toISOString(),
        metadata: r.metadata ?? undefined,
      }));

      return { events, total };
    });
  }
}
```

```typescript
    dateRange: query.dateRange,

  };
```

```typescript
function AuditLogContent(): React.ReactElement {
  const [events, setEvents] = useState<AuditEventDto[]>([]);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(true);
  const [loadingMore, setLoadingMore] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [dateFilter, setDateFilter] = useState<string>('30d');
  const [categoryFilter, setCategoryFilter] = useState<string>('');

  const fetchPage = useCallback(async (p: number, append = false): Promise<void> => {
    if (p === 1) setLoading(true);
    else setLoadingMore(true);
    setError(null);
    try {
      const params = new URLSearchParams({
        page: String(p),
        pageSize: String(PAGE_SIZE),
        dateRange: dateFilter,
        ...(categoryFilter ? { category: categoryFilter } : {}),
      });
      const res = await api.get<AuditLogResponse>(`/api/v1/auth/audit-log?${params.toString()}`);
      const data = res.data;
      setTotal(data.total);
      setEvents((prev) => (append ? [...prev, ...data.events] : data.events));
    } catch {
      setError('गतिविधि लॉड नहीं हो सका। दोबारा कोशिश करें।');
    } finally {
      setLoading(false);
      setLoadingMore(false);
    }
  }, [dateFilter, categoryFilter]);

  useEffect(() => {
    setPage(1);
    setEvents([]);
    void fetchPage(1, false);
  }, [fetchPage]);

  const handleEndReached = (): void => {
    const hasMore = events.length < total;
    if (!loadingMore && hasMore) {
      const nextPage = page + 1;
      setPage(nextPage);
      void fetchPage(nextPage, true);
    }
  };

  if (loading) {
    return (
      <View style={styles.container}>
        <Skeleton height={56} style={styles.skeletonRow} />
        <Skeleton height={56} style={styles.skeletonRow} />
        <Skeleton height={56} style={styles.skeletonRow} />
      </View>
    );
  }

  if (error !== null && events.length === 0) {
    return (
      <View style={styles.centeredPad}>
        <Text style={styles.errorText}>{error}</Text>
        <TouchableOpacity
          style={styles.retryButton}
          onPress={() => void fetchPage(1, false)}
          accessibilityRole="button"
        >
          <Text style={styles.retryButtonText}>फिर से लोड करें</Text>
        </TouchableOpacity>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      {/* Date filter chips */}
      <ScrollView
        horizontal
        showsHorizontalScrollIndicator={false}
        style={styles.filterRow}
        contentContainerStyle={styles.filterRowContent}
      >
        {DATE_FILTERS.map((f) => (
          <TouchableOpacity
            key={f.key}
            testID={`filter-date-${f.key}`}
            style={[styles.chip, dateFilter === f.key && styles.chipActive]}
            onPress={() => setDateFilter(f.key)}
            accessibilityRole="button"
            accessibilityState={{ selected: dateFilter === f.key }}
          >
            <Text style={[styles.chipText, dateFilter === f.key && styles.chipTextActive]}>
              {f.label}
            </Text>
          </TouchableOpacity>
        ))}
      </ScrollView>

      {/* Category filter chips */}
      <ScrollView
        horizontal
        showsHorizontalScrollIndicator={false}
        style={styles.filterRow}
        contentContainerStyle={styles.filterRowContent}
      >
        {CATEGORY_FILTERS.map((f) => (
          <TouchableOpacity
            key={f.key || 'all'}
            testID={`filter-category-${f.key || 'all'}`}
            style={[styles.chip, categoryFilter === f.key && styles.chipActive]}
            onPress={() => setCategoryFilter(f.key)}
            accessibilityRole="button"
            accessibilityState={{ selected: categoryFilter === f.key }}
          >
            <Text style={[styles.chipText, categoryFilter === f.key && styles.chipTextActive]}>
              {f.label}
            </Text>
          </TouchableOpacity>
        ))}
      </ScrollView>

      {/* Event list */}
      <FlatList
        data={events}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <AuditEventRow
            action={item.action}
            actorName={item.actorName}
            actorRole={item.actorRole}
            createdAt={item.createdAt}
            metadata={item.metadata}
            testID={`audit-event-${item.id}`}
          />
        )}
        onEndReached={handleEndReached}
        onEndReachedThreshold={0.3}
        ListEmptyComponent={
          <View style={styles.centeredPad}>
            <Text style={styles.emptyText}>कोई गतिविधि नहीं</Text>
          </View>
        }
        ListFooterComponent={
          loadingMore ? (
            <ActivityIndicator size="small" color={colors.primary} style={{ marginVertical: spacing.md }} />
          ) : null
        }
      />
    </View>
  );
}
```

# Story 2.4: Loyalty Program Configuration — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `GET + PATCH /api/v1/settings/loyalty` endpoints and a mobile `LoyaltyTierForm` component so shopkeepers can configure 3 loyalty tiers (name, threshold in rupees, badge colour) and earn/redemption rates.

**Architecture:** Shared Zod schema in `packages/shared` owns `LoyaltyConfig`, `PatchLoyaltySchema` (discriminated union `type: "tier" | "rate"`), and `LOYALTY_DEFAULTS`. `SettingsCache` grows per-key loyalty methods (`shop:{id}:settings:loyalty`, TTL 60s). Service converts `thresholdRupees` string → integer paise, enforces tier ordering (422 if violated), upserts the full `LoyaltyConfig` into `loyalty_json` JSONB (column exists from migration 0006 — no migration needed). Rates are stored as DECIMAL strings, never as floats.

**Tech Stack:** TypeScript strict, Zod 3, NestJS 10, PostgreSQL 15 (`withTenantTx`), Vitest, `@testcontainers/postgresql`, React Native (Expo), NativeWind, `@testing-library/react`.

**Paise constants:**
- `MAX_THRESHOLD_PAISE = 1_000_000_000` (₹1 crore = 10^9 paise)
- `MAX_THRESHOLD_RUPEES = "10000000"` (₹1 crore = 10^7 rupees)
- Conversion: `Math.round(parseFloat(rupeesStr) * 100)` — always an integer; validate ≥ 0 and ≤ `MAX_THRESHOLD_PAISE`

```typescript
import { z } from 'zod';

const HEX_COLOR_REGEX = /^#[0-9A-Fa-f]{6}$/;

const MAX_THRESHOLD_PAISE = 1_000_000_000; // ₹1 crore in paise

const positiveDecimalString = z
  .string()
  .regex(/^\d+(\.\d+)?$/, 'VALUE_FORMAT_INVALID')
  .refine((s) => parseFloat(s) > 0 && parseFloat(s) <= 100, 'RATE_OUT_OF_RANGE');

export const LoyaltyTierSchema = z.object({
  name:           z.string().min(1, 'TIER_NAME_REQUIRED').max(20, 'TIER_NAME_TOO_LONG'),
  thresholdPaise: z.number().int('THRESHOLD_MUST_BE_INTEGER').min(0, 'THRESHOLD_MIN').max(MAX_THRESHOLD_PAISE, 'THRESHOLD_MAX'),
  badgeColor:     z.string().regex(HEX_COLOR_REGEX, 'BADGE_COLOR_INVALID'),
});
export type LoyaltyTier = z.infer<typeof LoyaltyTierSchema>;

export const LoyaltyConfigSchema = z.object({
  tiers:                    z.tuple([LoyaltyTierSchema, LoyaltyTierSchema, LoyaltyTierSchema]),
  earnRatePercentage:       positiveDecimalString,
  redemptionRatePercentage: positiveDecimalString,
});
export type LoyaltyConfig = z.infer<typeof LoyaltyConfigSchema>;

const thresholdRupeesString = z
  .string()
  .regex(/^\d+(\.\d{1,2})?$/, 'VALUE_FORMAT_INVALID')
  .refine((s) => parseFloat(s) >= 0 && parseFloat(s) <= 10_000_000, 'THRESHOLD_OUT_OF_RANGE');

export const PatchLoyaltyTierSchema = z.object({
  type:            z.literal('tier'),
  index:           z.union([z.literal(0), z.literal(1), z.literal(2)]),
  name:            z.string().min(1, 'TIER_NAME_REQUIRED').max(20, 'TIER_NAME_TOO_LONG'),
  thresholdRupees: thresholdRupeesString,
  badgeColor:      z.string().regex(HEX_COLOR_REGEX, 'BADGE_COLOR_INVALID'),
});
export type PatchLoyaltyTierDto = z.infer<typeof PatchLoyaltyTierSchema>;

export const PatchLoyaltyRateSchema = z.object({
  type:                     z.literal('rate'),
  earnRatePercentage:       positiveDecimalString,
  redemptionRatePercentage: positiveDecimalString,
});
export type PatchLoyaltyRateDto = z.infer<typeof PatchLoyaltyRateSchema>;

export const PatchLoyaltySchema = z.discriminatedUnion('type', [
  PatchLoyaltyTierSchema,
  PatchLoyaltyRateSchema,
]);
export type PatchLoyaltyDto = z.infer<typeof PatchLoyaltySchema>;

export const LOYALTY_DEFAULTS: LoyaltyConfig = {
  tiers: [
    { name: 'Silver',  thresholdPaise: 5_000_000,  badgeColor: '#C0C0C0' },
    { name: 'Gold',    thresholdPaise: 15_000_000,  badgeColor: '#FFD700' },
    { name: 'Diamond', thresholdPaise: 50_000_000,  badgeColor: '#B9F2FF' },
  ],
  earnRatePercentage:       '1.00',
  redemptionRatePercentage: '1.00',
};
```

```typescript
  async getLoyalty(): Promise<LoyaltyConfig | null> {
    const key = this.loyaltyKey();
    try {
      const raw = await this.redis.get(key);
      if (!raw) return null;
      const parsed: unknown = JSON.parse(raw);
      const result = LoyaltyConfigSchema.safeParse(parsed);
      if (!result.success) {
        await this.redis.del(key);
        return null;
      }
      return result.data;
    } catch {
      try { await this.redis.del(key); } catch { /* ignore */ }
      return null;
    }
  }

  async setLoyalty(data: LoyaltyConfig): Promise<void> {
    await this.redis.set(this.loyaltyKey(), JSON.stringify(data), 'EX', this.ttlSec);
  }

  async invalidateLoyalty(): Promise<void> {
    await this.redis.del(this.loyaltyKey());
  }

  private loyaltyKey(): string {
    return `shop:${tenantContext.requireCurrent().shopId}:settings:loyalty`;
  }
```

```typescript
  async upsertLoyalty(config: LoyaltyConfig): Promise<UpdateLoyaltyResult> {
    return withTenantTx(this.pool, async (tx) => {
      const shopId = tenantContext.requireCurrent().shopId;

      // Ensure row exists so SELECT FOR UPDATE can lock it.
      await tx.query(
        `INSERT INTO shop_settings (shop_id) VALUES ($1) ON CONFLICT (shop_id) DO NOTHING`,
        [shopId],
      );

      const beforeRow = await tx.query<{ loyalty_json: LoyaltyConfig | null }>(
        `SELECT loyalty_json FROM shop_settings WHERE shop_id = $1 FOR UPDATE`,
        [shopId],
      );
      const before = beforeRow.rows[0]?.loyalty_json ?? null;

      const r = await tx.query<{ loyalty_json: LoyaltyConfig }>(
        `INSERT INTO shop_settings (shop_id, loyalty_json)
         VALUES ($1, $2::jsonb)
         ON CONFLICT (shop_id)
         DO UPDATE SET loyalty_json = $2::jsonb, updated_at = now()
         RETURNING loyalty_json`,
        [shopId, JSON.stringify(config)],
      );

      return { before, after: r.rows[0].loyalty_json };
    });
  }
```

```typescript
function rupeesToPaise(rupeesStr: string): number {
  const n = parseFloat(rupeesStr);
  const paise = Math.round(n * 100);
  if (paise < 0 || paise > 1_000_000_000) {
    throw new UnprocessableEntityException({ code: 'settings.threshold_out_of_range' });
  }
  return paise;
}
```

```typescript
  async updateLoyalty(dto: PatchLoyaltyDto): Promise<LoyaltyConfig> {
    const parsed = PatchLoyaltySchema.safeParse(dto);
    if (!parsed.success) {
      const errors = parsed.error.issues.map((i) => ({ field: i.path.join('.'), code: i.message }));
      throw new BadRequestException({ code: 'validation.failed', errors });
    }

    const current = await this.getLoyalty();
    let updated: LoyaltyConfig;

    if (parsed.data.type === 'tier') {
      const { index, name, thresholdRupees, badgeColor } = parsed.data;
      const newTiers = [...current.tiers] as LoyaltyConfig['tiers'];
      newTiers[index] = { name, thresholdPaise: rupeesToPaise(thresholdRupees), badgeColor };

      if (
        newTiers[0].thresholdPaise >= newTiers[1].thresholdPaise ||
        newTiers[1].thresholdPaise >= newTiers[2].thresholdPaise
      ) {
        throw new UnprocessableEntityException({
          code: 'settings.tier_order_invalid',
          message: 'Tier thresholds बढ़ते क्रम में होने चाहिए',
        });
      }

      updated = { ...current, tiers: newTiers };
    } else {
      updated = {
        ...current,
        earnRatePercentage: parsed.data.earnRatePercentage,
        redemptionRatePercentage: parsed.data.redemptionRatePercentage,
      };
    }

    const { before, after } = await this.repo.upsertLoyalty(updated);

    await this.cache.invalidateLoyalty();

    void this.auditLoyaltyUpdate(before, after).catch(() => undefined);

    return after;
  }

  private async auditLoyaltyUpdate(
    before: LoyaltyConfig | null,
    after: LoyaltyConfig,
  ): Promise<void> {
    const tc = tenantContext.current();
    if (!tc) return;
    await auditLog(this.pool, {
      action: AuditAction.SETTINGS_LOYALTY_UPDATED,
      subjectType: 'shop',
      subjectId: tc.shopId,
      actorUserId: tc.authenticated ? tc.userId : undefined,
      before,
      after,
    });
  }
```

```typescript
import React, { useState, useEffect } from 'react';
import { View, Text, TextInput, StyleSheet } from 'react-native';
import { t } from '@goldsmith/i18n';
import type { LoyaltyConfig, LoyaltyTier, PatchLoyaltyDto } from '@goldsmith/shared';
import { colors, spacing, typography } from '@goldsmith/ui-tokens';

interface Props {
  config: LoyaltyConfig;
  onBlurSave: (dto: PatchLoyaltyDto) => Promise<void>;
  saving: boolean;
  error: string | null;
}

function paiseToRupeesStr(paise: number): string {
  return (paise / 100).toFixed(2);
}

function TierRow({
  tier,
  index,
  onBlurSave,
}: {
  tier: LoyaltyTier;
  index: 0 | 1 | 2;
  onBlurSave: (dto: PatchLoyaltyDto) => Promise<void>;
}): React.ReactElement {
  const [name, setName] = useState(tier.name);
  const [threshold, setThreshold] = useState(paiseToRupeesStr(tier.thresholdPaise));
  const [localError, setLocalError] = useState<string | null>(null);

  useEffect(() => { setName(tier.name); }, [tier.name]);
  useEffect(() => { setThreshold(paiseToRupeesStr(tier.thresholdPaise)); }, [tier.thresholdPaise]);

  async function handleFieldBlur(): Promise<void> {
    setLocalError(null);
    try {
      await onBlurSave({
        type: 'tier',
        index,
        name,
        thresholdRupees: threshold,
        badgeColor: tier.badgeColor,
      });
    } catch (err: unknown) {
      const code = (err as { response?: { code?: string } })?.response?.code;
      if (code === 'settings.tier_order_invalid') {
        setLocalError(t('settings.loyalty.errors.tier_order_invalid'));
      }
    }
  }

  return (
    <View testID={`tier-row-${index}`} style={styles.tierRow}>
      <View
        testID={`badge-circle-${index}`}
        style={[styles.badge, { backgroundColor: tier.badgeColor }]}
      />
      <TextInput
        testID={`name-input-${index}`}
        value={name}
        onChangeText={setName}
        onBlur={handleFieldBlur}
        style={styles.nameInput}
        maxLength={20}
        accessibilityLabel={t('settings.loyalty.tier_threshold_label')}
      />
      <TextInput
        testID={`threshold-input-${index}`}
        value={threshold}

        onBlur={handleFieldBlur}
        keyboardType="decimal-pad"
        style={styles.thresholdInput}
        accessibilityLabel={t('settings.loyalty.tier_threshold_label')}
      />
      {localError !== null && (
        <Text style={styles.errorText}>{localError}</Text>
      )}
    </View>
  );
}

    await onBlurSave({
      type: 'rate',
      earnRatePercentage: earnRate,
      redemptionRatePercentage: redemptionRate,
    });
  }

  return (
    <View>
      {(config.tiers as [typeof config.tiers[0], typeof config.tiers[0], typeof config.tiers[0]]).map((tier, i) => (
        <TierRow
          key={i}
          tier={tier}
          index={i as 0 | 1 | 2}
          onBlurSave={onBlurSave}
        />
      ))}

      <View style={styles.rateRow}>
        <Text style={styles.rateLabel}>{t('settings.loyalty.earn_rate_label')}</Text>
        <Text style={styles.rateHint}>
          {t('settings.loyalty.earn_rate_hint').replace('{rate}', earnRate)}
        </Text>
        <TextInput
          testID="earn-rate-input"
          value={earnRate}
          onChangeText={setEarnRate}
          onBlur={handleRateBlur}
          keyboardType="decimal-pad"
          style={styles.rateInput}
        />
      </View>

      <View style={styles.rateRow}>
        <Text style={styles.rateLabel}>{t('settings.loyalty.redemption_rate_label')}</Text>
        <Text style={styles.rateHint}>
          {t('settings.loyalty.redemption_hint').replace('{rate}', redemptionRate)}

  );
}
```

```typescript
import React, { useState, useCallback } from 'react';
import { ScrollView, StyleSheet } from 'react-native';
import { LoyaltyTierForm } from '../../src/features/settings/components/LoyaltyTierForm';
import type { LoyaltyConfig, PatchLoyaltyDto } from '@goldsmith/shared';
import { LOYALTY_DEFAULTS } from '@goldsmith/shared';
import { colors, spacing } from '@goldsmith/ui-tokens';

// TODO: replace with real TanStack Query hook once API client is wired
export default function LoyaltyScreen(): React.ReactElement {
  const [config, setConfig] = useState<LoyaltyConfig>(LOYALTY_DEFAULTS);
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleBlurSave = useCallback(async (dto: PatchLoyaltyDto): Promise<void> => {
    setSaving(true);
    setError(null);
    try {
      const res = await fetch('/api/v1/settings/loyalty', {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(dto),
      });
      if (!res.ok) {
        const body = (await res.json()) as { error?: { code?: string; message?: string } };
        setError(body.error?.message ?? 'Unknown error');
        throw body;
      }
      const updated = (await res.json()) as LoyaltyConfig;
      setConfig(updated);
    } finally {
      setSaving(false);
    }
  }, []);

  return (
    <ScrollView style={styles.container}>
      <LoyaltyTierForm
        config={config}
        onBlurSave={handleBlurSave}
        saving={saving}
        error={error}
      />
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: colors.bg,
    paddingVertical: spacing.sm,
  },
});
```

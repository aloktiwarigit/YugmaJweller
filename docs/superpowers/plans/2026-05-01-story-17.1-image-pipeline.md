# Story 17.1 — Product Image Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the customer-facing `GoldTexturePlaceholder` stub with a real product-image pipeline. Shopkeepers upload, reorder, edit alt-text, and delete photos against a product. Customer storefront renders a multi-image gallery via ImageKit-transformed CDN URLs with `mb-0.25` byte cap, EXIF-stripped originals, and tenant-RLS-isolated metadata.

**Architecture:** Server-routed multipart upload (5 MB cap) → `sharp` 1920w probe → EXIF strip via `.rotate()` (default-strips metadata) → Azure Blob (or stub) write → `withTenantTx` with pessimistic `SELECT ... FOR UPDATE` on product (tenant-ownership check + 10-cap enforcement + insert + audit, all serialized). Read path: shopkeeper sees `ImageRow` (with `storage_key`); customers see `PublicImageRow` (server-built `srcset` + `default_url` + `placeholder_url`, no `storage_key`). ImageKit transforms enforce 250 KB via `mb-0.25` on the CDN side; sharp probe is the upload-time pre-reject. Retires the legacy `InventoryService.getImageUploadUrl` fire-and-forget path.

**Tech Stack:** NestJS, PostgreSQL RLS, `sharp` (libvips), `@azure/storage-blob`, ImageKit transform URLs, `file-type` magic-byte sniffer, `expo-image-picker`, `react-native-draggable-flatlist`, Next.js 14 App Router, vitest.

```sql
  updated_at           TIMESTAMPTZ NOT NULL DEFAULT now()
```

```typescript
import { promises as fs } from 'node:fs';
import * as path from 'node:path';
import type { StoragePort } from '../storage.port';

const PORT = process.env['PORT'] ?? '3000';

/**
 * Dev/CI stub: writes blobs to local disk under STUB_STORAGE_DIR (default ./tmp/storage).
 * Public URLs route to a dev-only Express middleware on the API at /dev-storage/:key
 * that serves files from STUB_STORAGE_DIR. Bound to 127.0.0.1 only — never deployed.
 */
export class StubStorageAdapter implements StoragePort {
  private readonly baseDir: string;
  private readonly publicBase: string;

  constructor() {
    this.baseDir = process.env['STUB_STORAGE_DIR'] ?? path.join(process.cwd(), 'tmp', 'storage');
    this.publicBase = process.env['STUB_STORAGE_PUBLIC_BASE'] ?? `http://127.0.0.1:${PORT}/dev-storage`;
  }

  async getPresignedUploadUrl(key: string, _contentType: string): Promise<string> {
    return `${this.publicBase}/${key}?upload=STUB`;
  }

  async getPublicUrl(key: string): Promise<string> {
    return `${this.publicBase}/${key}`;
  }

  async downloadBuffer(key: string): Promise<Buffer> {
    const absPath = path.join(this.baseDir, key);
    return fs.readFile(absPath);
  }

  async uploadBuffer(key: string, data: Buffer, _contentType: string): Promise<void> {
    const absPath = path.join(this.baseDir, key);
    await fs.mkdir(path.dirname(absPath), { recursive: true });
    await fs.writeFile(absPath, data);
  }

  async getPresignedReadUrl(key: string): Promise<string> {
    return this.getPublicUrl(key);
  }

  async deleteBlob(key: string): Promise<void> {
    const absPath = path.join(this.baseDir, key);
    try {
      await fs.unlink(absPath);
    } catch (err: unknown) {
      // Best-effort: ignore missing file (matches Azure adapter's leniency)
      if ((err as NodeJS.ErrnoException).code !== 'ENOENT') throw err;
    }
  }
}
```

```typescript
import {
  BlobServiceClient,
  StorageSharedKeyCredential,
  generateBlobSASQueryParameters,
  BlobSASPermissions,
} from '@azure/storage-blob';
import type { StoragePort } from '../storage.port';

const SAS_TTL_MINUTES_UPLOAD = 60;     // 1 hour for upload SAS (one-shot)
const SAS_TTL_MINUTES_READ   = 60;     // 1 hour for read SAS (used by reconciliation only)

/**
 * Production adapter against Azure Blob Storage (Central or South India per ADR-0015).
 * Originals are stored privately; ImageKit Web Folder fetches them server-side
 * with read-only credentials configured in the ImageKit dashboard.
 *
 * Selected when STORAGE_ADAPTER=azure-imagekit. STUB is the default for dev/CI.
 */
export class AzureBlobAdapter implements StoragePort {
  private readonly accountName: string;
  private readonly containerName: string;
  private readonly credential: StorageSharedKeyCredential;
  private readonly serviceClient: BlobServiceClient;

  constructor() {
    this.accountName   = required('AZURE_STORAGE_ACCOUNT');
    this.containerName = required('AZURE_STORAGE_CONTAINER');
    const accountKey   = required('AZURE_STORAGE_ACCOUNT_KEY');
    this.credential = new StorageSharedKeyCredential(this.accountName, accountKey);
    this.serviceClient = new BlobServiceClient(
      `https://${this.accountName}.blob.core.windows.net`,
      this.credential,
    );
  }

  async getPresignedUploadUrl(key: string, contentType: string): Promise<string> {
    const sas = generateBlobSASQueryParameters(
      {
        containerName: this.containerName,
        blobName: key,
        permissions: BlobSASPermissions.parse('cw'),
        startsOn: new Date(Date.now() - 60_000),                                // 1-min skew
        expiresOn: new Date(Date.now() + SAS_TTL_MINUTES_UPLOAD * 60_000),
        contentType,
      },
      this.credential,
    ).toString();
    return `https://${this.accountName}.blob.core.windows.net/${this.containerName}/${key}?${sas}`;
  }

  async getPublicUrl(key: string): Promise<string> {
    // Public URL goes through ImageKit Web Folder, NOT Azure directly.
    // ImageKit fetches the private Azure blob using credentials configured
    // in the ImageKit dashboard. Customers never receive Azure URLs.
    const imagekitId = required('IMAGEKIT_ID');
    return `https://ik.imagekit.io/${imagekitId}/${key}`;
  }

  async downloadBuffer(key: string): Promise<Buffer> {
    const blob = this.serviceClient.getContainerClient(this.containerName).getBlockBlobClient(key);
    return await blob.downloadToBuffer();
  }

  async uploadBuffer(key: string, data: Buffer, contentType: string): Promise<void> {
    const blob = this.serviceClient.getContainerClient(this.containerName).getBlockBlobClient(key);
    await blob.uploadData(data, { blobHTTPHeaders: { blobContentType: contentType } });
  }

  async getPresignedReadUrl(key: string): Promise<string> {
    const sas = generateBlobSASQueryParameters(
      {
        containerName: this.containerName,
        blobName: key,
        permissions: BlobSASPermissions.parse('r'),
        startsOn: new Date(Date.now() - 60_000),
        expiresOn: new Date(Date.now() + SAS_TTL_MINUTES_READ * 60_000),
      },
      this.credential,
    ).toString();
    return `https://${this.accountName}.blob.core.windows.net/${this.containerName}/${key}?${sas}`;
  }

  async deleteBlob(key: string): Promise<void> {
    const blob = this.serviceClient.getContainerClient(this.containerName).getBlockBlobClient(key);
    try {
      await blob.delete();
    } catch (err: unknown) {
      const status = (err as { statusCode?: number }).statusCode;
      if (status !== 404) throw err;     // 404 = already deleted; lenient
    }
  }
}

function required(name: string): string {
  const v = process.env[name];
  if (!v) throw new Error(`AzureBlobAdapter requires env var ${name}`);
  return v;
}
```

```typescript
/**
 * Pure URL builder. No HTTP client, no auth. Every URL emitted carries
 * `q-auto`, `f-auto`, and `mb-0.25` — the binding NFR-IMG-1 enforcement.
 * No code path may compose ImageKit URLs by hand; all callers go through
 * this builder.
 */
export class ImageKitTransformUrlBuilder {
  private readonly base: string;

  constructor() {
    const id = process.env['IMAGEKIT_ID'] ?? 'goldsmith';
    this.base = `https://ik.imagekit.io/${id}`;
  }

  url(key: string, opts: { width: number; blur?: number }): string {
    const parts: string[] = [`w-${opts.width}`];
    if (opts.blur !== undefined) parts.push(`bl-${opts.blur}`);
    parts.push('q-auto', 'f-auto', 'mb-0.25');
    return `${this.base}/${key}?tr=${parts.join(',')}`;
  }

  /**
   * Builds the customer-side srcset string for the 4 standard widths.
   * Output: "url320 320w, url640 640w, url1024 1024w, url1920 1920w"
   */
  srcset(key: string): string {
    return [320, 640, 1024, 1920]
      .map((w) => `${this.url(key, { width: w })} ${w}w`)
      .join(', ');
  }
}

export const IMAGEKIT_URL_BUILDER = 'IMAGEKIT_URL_BUILDER';
```

```typescript
import { Inject, Injectable } from '@nestjs/common';
import type { Pool, PoolClient } from 'pg';
import { withTenantTx } from '../../db/with-tenant-tx';

type Tx = Pick<PoolClient, 'query'>;

export type ImageRow = {
  id: string;
  shop_id: string;
  product_id: string;
  storage_key: string;
  alt_text: string | null;
  mime_type: string;
  byte_size: number;
  width: number;
  height: number;
  exif_stripped_at: string;
  uploaded_by_user_id: string;
  scan_status: 'pending' | 'clean' | 'rejected';
  sort_order: number;
  created_at: string;
  updated_at: string;
};

export type InsertImageInput = {
  shopId: string;
  productId: string;
  storageKey: string;
  mimeType: string;
  byteSize: number;
  width: number;
  height: number;
  sortOrder: number;
  altText: string | null;
  uploadedByUserId: string;
};

const SELECT_COLS = `
  id, shop_id, product_id, storage_key, alt_text, mime_type, byte_size,
  width, height, exif_stripped_at, uploaded_by_user_id, scan_status,
  sort_order, created_at, updated_at
`.trim();

@Injectable()
export class ProductImagesRepository {
  constructor(@Inject('PG_POOL') private readonly pool: Pool) {}

  /**
   * Tenant-scoped product existence check + pessimistic row lock.
   * Returns null on miss (cross-tenant attempt OR not found).
   *
   * The FOR UPDATE lock serializes concurrent uploads for the same product,
   * making the 10-image cap inviolable. SELECT alone is insufficient because
   * PostgreSQL FK checks bypass RLS — this is the explicit security layer.
   */
  async lockProductForTenant(tx: Tx, shopId: string, productId: string): Promise<{ id: string } | null> {
    const r = await tx.query<{ id: string }>(
      `SELECT id FROM products WHERE id = $1 AND shop_id = $2 FOR UPDATE`,
      [productId, shopId],
    );
    return r.rows[0] ?? null;
  }

  async countImagesInTx(tx: Tx, productId: string): Promise<number> {
    const r = await tx.query<{ count: string }>(
      `SELECT COUNT(*)::text AS count FROM product_images WHERE product_id = $1`,
      [productId],
    );
    return parseInt(r.rows[0]?.count ?? '0', 10);
  }

  async nextSortOrderInTx(tx: Tx, productId: string): Promise<number> {
    const r = await tx.query<{ next: number }>(
      `SELECT COALESCE(MAX(sort_order), -1) + 1 AS next FROM product_images WHERE product_id = $1`,
      [productId],
    );
    return r.rows[0]?.next ?? 0;
  }

  async insertImageInTx(tx: Tx, input: InsertImageInput): Promise<ImageRow> {
    const r = await tx.query<ImageRow>(
      `INSERT INTO product_images
         (shop_id, product_id, storage_key, alt_text, mime_type, byte_size,
          width, height, exif_stripped_at, uploaded_by_user_id, scan_status, sort_order)
       VALUES
         ($1, $2, $3, $4, $5, $6, $7, $8, NOW(), $9, 'clean', $10)
       RETURNING ${SELECT_COLS}`,
      [
        input.shopId, input.productId, input.storageKey, input.altText,
        input.mimeType, input.byteSize, input.width, input.height,
        input.uploadedByUserId, input.sortOrder,
      ],
    );
    return r.rows[0]!;
  }

  async listForProduct(shopId: string, productId: string): Promise<ImageRow[]> {
    return withTenantTx(this.pool, async (tx) => {
      const r = await tx.query<ImageRow>(
        `SELECT ${SELECT_COLS}
           FROM product_images
          WHERE product_id = $1 AND shop_id = $2
          ORDER BY sort_order ASC, created_at ASC`,
        [productId, shopId],
      );
      return r.rows;
    });
  }

  async deleteImage(shopId: string, productId: string, imageId: string): Promise<{ storageKey: string } | null> {
    return withTenantTx(this.pool, async (tx) => {
      const r = await tx.query<{ storage_key: string }>(
        `DELETE FROM product_images
          WHERE id = $1 AND product_id = $2 AND shop_id = $3
          RETURNING storage_key`,
        [imageId, productId, shopId],
      );
      const row = r.rows[0];
      return row ? { storageKey: row.storage_key } : null;
    });
  }

  async setSortOrders(shopId: string, productId: string, orderedIds: string[]): Promise<ImageRow[]> {
    return withTenantTx(this.pool, async (tx) => {
      // Verify the set matches exactly: every image of the product is in
      // orderedIds and no extras. Mismatch → ORDER_LIST_MISMATCH (caller throws 400).
      const existing = await tx.query<{ id: string }>(
        `SELECT id FROM product_images WHERE product_id = $1 AND shop_id = $2 FOR UPDATE`,
        [productId, shopId],
      );
      const existingSet = new Set(existing.rows.map((r) => r.id));
      const orderedSet = new Set(orderedIds);
      if (existingSet.size !== orderedSet.size) return [];
      for (const id of existingSet) if (!orderedSet.has(id)) return [];

      // Atomic update: each row gets its new sort_order based on array index.
      for (let i = 0; i < orderedIds.length; i++) {
        await tx.query(
          `UPDATE product_images
              SET sort_order = $1, updated_at = NOW()
            WHERE id = $2 AND product_id = $3 AND shop_id = $4`,
          [i, orderedIds[i], productId, shopId],
        );
      }
      const r = await tx.query<ImageRow>(
        `SELECT ${SELECT_COLS}
           FROM product_images
          WHERE product_id = $1 AND shop_id = $2
          ORDER BY sort_order ASC`,
        [productId, shopId],
      );
      return r.rows;
    });
  }

  async setAltText(shopId: string, productId: string, imageId: string, altText: string | null): Promise<ImageRow | null> {
    return withTenantTx(this.pool, async (tx) => {
      const r = await tx.query<ImageRow>(
        `UPDATE product_images
            SET alt_text = $1, updated_at = NOW()
          WHERE id = $2 AND product_id = $3 AND shop_id = $4
          RETURNING ${SELECT_COLS}`,
        [altText, imageId, productId, shopId],
      );
      return r.rows[0] ?? null;
    });
  }
}
```

```typescript
import {
  BadRequestException, ConflictException, Inject, Injectable, NotFoundException, PayloadTooLargeException,
} from '@nestjs/common';
import { fileTypeFromBuffer } from 'file-type';
import sharp from 'sharp';
import type { Pool } from 'pg';
import { randomUUID } from 'node:crypto';
import { withTenantTx } from '../../db/with-tenant-tx';
import { AuditAction, type AuditService } from '@goldsmith/audit';
import { STORAGE_PORT } from '@goldsmith/integrations-storage';
import type { StoragePort, MalwareScanPort } from '@goldsmith/integrations-storage';
import { MALWARE_SCAN_PORT } from '@goldsmith/integrations-storage';
import { ProductImagesRepository, type ImageRow } from './product-images.repository';

const MAX_BYTES        = 5 * 1024 * 1024;          // 5 MB
const PROBE_WIDTH      = 1920;
const PROBE_QUALITY    = 80;
const PROBE_EFFORT     = 6;
const VARIANT_BYTE_CAP = 250_000;                   // 250 KB
const MIN_DIM          = 200;
const MAX_DIM          = 8000;
const MAX_IMAGES_PER_PRODUCT = 10;

const ALLOW_LIST = new Set(['image/jpeg', 'image/png', 'image/webp', 'image/heic']);

export type UploadInput = {
  shopId: string;
  productId: string;
  userId: string;
  file: { buffer: Buffer; mimeType: string; size: number };
  altText?: string | null;
};

@Injectable()
export class ProductImagesService {
  constructor(
    private readonly repo: ProductImagesRepository,
    @Inject(STORAGE_PORT) private readonly storage: StoragePort,
    @Inject(MALWARE_SCAN_PORT) private readonly malwareScan: MalwareScanPort,
    private readonly audit: AuditService,
    @Inject('PG_POOL') private readonly pool: Pool,
  ) {}

  async upload(input: UploadInput): Promise<ImageRow> {
    const { shopId, productId, userId, file, altText } = input;

    // 1. Body-size cap (before any sharp work — defense against libvips CVEs)
    if (file.size > MAX_BYTES) {
      throw new PayloadTooLargeException({ code: 'PAYLOAD_TOO_LARGE' });
    }

    // 2. MIME sniff via magic bytes; reject SVG outright
    const sniffed = await fileTypeFromBuffer(file.buffer);
    if (!sniffed || !ALLOW_LIST.has(sniffed.mime) || sniffed.mime === 'image/svg+xml') {
      await this.audit.emit(undefined, AuditAction.PRODUCT_IMAGE_REJECTED, {
        productId, reason: 'INVALID_MIME', detectedMime: sniffed?.mime ?? 'unknown',
      });
      throw new BadRequestException({ code: 'INVALID_MIME' });
    }

    // 3. Dimension guards
    const meta = await sharp(file.buffer).metadata();
    if (!meta.width || !meta.height) throw new BadRequestException({ code: 'INVALID_DIMENSIONS' });
    if (meta.width < MIN_DIM || meta.height < MIN_DIM) throw new BadRequestException({ code: 'INVALID_DIMENSIONS' });
    if (meta.width > MAX_DIM || meta.height > MAX_DIM) throw new BadRequestException({ code: 'INVALID_DIMENSIONS' });

    // 4. Variant byte-cap probe (worst-case width = 1920)
    const probe = await sharp(file.buffer)
      .rotate()
      .resize({ width: PROBE_WIDTH, withoutEnlargement: true })
      .toFormat('webp', { quality: PROBE_QUALITY, effort: PROBE_EFFORT })
      .toBuffer();
    if (probe.byteLength > VARIANT_BYTE_CAP) {
      await this.audit.emit(undefined, AuditAction.PRODUCT_IMAGE_REJECTED, {
        productId, reason: 'IMAGE_TOO_LARGE_AFTER_COMPRESSION',
        probeByteSize: probe.byteLength,
      });
      throw new BadRequestException({ code: 'IMAGE_TOO_LARGE_AFTER_COMPRESSION' });
    }

    // 5. EXIF strip via .rotate() (sharp default toBuffer strips remaining metadata)
    const cleaned = await sharp(file.buffer).rotate().toBuffer();

    // 5b. Re-read metadata AFTER rotation (orientation 6/8 swaps W/H)
    const cleanedMeta = await sharp(cleaned).metadata();
    if (!cleanedMeta.width || !cleanedMeta.height) {
      throw new BadRequestException({ code: 'INVALID_DIMENSIONS' });    // unreachable in practice
    }

    // 6. Malware scan (stub returns clean in MVP)
    const malware = await this.malwareScan.scan(cleaned, sniffed.mime);
    if (!malware.clean) {
      await this.audit.emit(undefined, AuditAction.PRODUCT_IMAGE_REJECTED, {
        productId, reason: 'SCAN_FAILED', detail: malware.reason,
      });
      throw new BadRequestException({ code: 'SCAN_FAILED' });
    }

    // 7. Storage upload (BEFORE DB tx; orphan on tx failure is acceptable)
    const ext = extFromMime(sniffed.mime);
    const storageKey = `tenant/${shopId}/products/${productId}/${randomUUID()}.${ext}`;
    await this.storage.uploadBuffer(storageKey, cleaned, sniffed.mime);

    // 8. Tx with pessimistic product-row lock — tenant ownership + cap + insert + audit serialize here
    try {
      return await withTenantTx(this.pool, async (tx) => {
        const owned = await this.repo.lockProductForTenant(tx, shopId, productId);
        if (!owned) {
          throw new NotFoundException({ code: 'PRODUCT_NOT_FOUND' });
        }

        const count = await this.repo.countImagesInTx(tx, productId);
        if (count >= MAX_IMAGES_PER_PRODUCT) {
          throw new ConflictException({ code: 'IMAGE_LIMIT_REACHED' });
        }

        const sortOrder = await this.repo.nextSortOrderInTx(tx, productId);
        const row = await this.repo.insertImageInTx(tx, {
          shopId, productId, storageKey,
          mimeType: sniffed.mime, byteSize: cleaned.length,
          width: cleanedMeta.width!, height: cleanedMeta.height!,
          sortOrder, altText: altText ?? null, uploadedByUserId: userId,
        });

        await this.audit.emit(tx, AuditAction.PRODUCT_IMAGE_UPLOADED, {
          imageId: row.id, productId, byteSize: cleaned.length, mimeType: sniffed.mime,
        });
        return row;
      });
    } catch (err) {
      // Best-effort orphan cleanup
      this.storage.deleteBlob(storageKey).catch(() => undefined);
      throw err;
    }
  }

  // Other methods (delete, listForProduct, reorder, setAltText) implemented in Task 5.
  async delete(_shopId: string, _productId: string, _imageId: string): Promise<void> { throw new Error('TODO Task 5'); }
  async listForProduct(_shopId: string, _productId: string): Promise<ImageRow[]> { throw new Error('TODO Task 5'); }
  async reorder(_shopId: string, _productId: string, _orderedIds: string[]): Promise<ImageRow[]> { throw new Error('TODO Task 5'); }
  async setAltText(_shopId: string, _productId: string, _imageId: string, _alt: string | null): Promise<ImageRow> { throw new Error('TODO Task 5'); }
}

function extFromMime(mime: string): string {
  switch (mime) {
    case 'image/jpeg': return 'jpg';
    case 'image/png':  return 'png';
    case 'image/webp': return 'webp';
    case 'image/heic': return 'heic';
    default: return 'bin';
  }
}
```

```typescript
  @HttpCode(201)

      productId,
      userId: ctx.userId,
      file: { buffer: file.buffer, mimeType: file.mimetype, size: file.size },
      altText: altText ?? null,
    });
  }

  @Roles('shop_admin', 'shop_manager')
  @HttpCode(204)

    await this.svc.delete(ctx.shopId, productId, imageId);
```

```typescript
import { useEffect, useState, useCallback, useRef } from 'react';
import { View, Text, Pressable, TextInput, Modal, ActivityIndicator, Alert } from 'react-native';
import * as ImagePicker from 'expo-image-picker';
import DraggableFlatList, { type RenderItemParams } from 'react-native-draggable-flatlist';
import { useLocalSearchParams } from 'expo-router';
import { useTranslation } from '@goldsmith/i18n';
import { api } from '../../../src/lib/api';
import { Image } from 'expo-image';

type ImageRow = {
  id: string;
  storage_key: string;
  alt_text: string | null;
  sort_order: number;
};

const IMAGEKIT_BASE = process.env.EXPO_PUBLIC_IMAGEKIT_ID
  ? `https://ik.imagekit.io/${process.env.EXPO_PUBLIC_IMAGEKIT_ID}`
  : 'https://ik.imagekit.io/goldsmith';

function thumbUrl(key: string): string {
  return `${IMAGEKIT_BASE}/${key}?tr=w-200,q-auto,f-auto,mb-0.25`;
}

export default function ProductImagesScreen() {
  const { id } = useLocalSearchParams<{ id: string }>();
  const { t } = useTranslation('inventory');
  const [images, setImages] = useState<ImageRow[]>([]);
  const [uploading, setUploading] = useState(false);
  const [confirmDeleteId, setConfirmDeleteId] = useState<string | null>(null);
  const mountedRef = useRef(true);

  useEffect(() => () => { mountedRef.current = false; }, []);

  const refresh = useCallback(async () => {
    const rows = await api.get<ImageRow[]>(`/products/${id}/images`);
    if (mountedRef.current) setImages(rows);
  }, [id]);

  useEffect(() => { void refresh(); }, [refresh]);

  const onAdd = async () => {
    const perm = await ImagePicker.requestMediaLibraryPermissionsAsync();
    if (!perm.granted) return;
    const picker = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: false,
      quality: 0.95,
    });
    if (picker.canceled) return;
    const asset = picker.assets[0];

    setUploading(true);
    try {
      const formData = new FormData();
      formData.append('file', { uri: asset.uri, name: asset.fileName ?? 'image.jpg', type: asset.mimeType ?? 'image/jpeg' } as never);
      await api.post(`/products/${id}/images`, formData, { headers: { 'Content-Type': 'multipart/form-data' } });
      await refresh();
    } catch (err: unknown) {
      const code = (err as { response?: { data?: { code?: string } } }).response?.data?.code ?? 'GENERIC';
      const key = `images_err_${code.toLowerCase()}`;
      Alert.alert(t(key) || t('images_err_generic'));
    } finally {
      if (mountedRef.current) setUploading(false);
    }
  };

  const onDelete = async (imageId: string) => {
    setConfirmDeleteId(null);
    try {
      await api.delete(`/products/${id}/images/${imageId}`);
      setImages((prev) => prev.filter((img) => img.id !== imageId));
    } catch {
      Alert.alert(t('images_err_generic'));
    }
  };

  const onAltChange = async (imageId: string, altText: string) => {
    try {
      await api.patch(`/products/${id}/images/${imageId}`, { alt_text: altText || null });
    } catch {
      Alert.alert(t('images_err_generic'));
    }
  };

  const onReorder = async (newOrder: ImageRow[]) => {
    setImages(newOrder);
    try {
      await api.patch(`/products/${id}/images/order`, { orderedIds: newOrder.map((r) => r.id) });
    } catch {
      void refresh();
      Alert.alert(t('images_err_generic'));
    }
  };

  const renderItem = ({ item, drag, isActive }: RenderItemParams<ImageRow>) => (
    <View style={{ flexDirection: 'row', alignItems: 'center', padding: 12, opacity: isActive ? 0.7 : 1 }}>
      <Image source={{ uri: thumbUrl(item.storage_key) }} style={{ width: 64, height: 64, borderRadius: 4 }} />
      <TextInput
        style={{ flex: 1, marginHorizontal: 12, fontSize: 16, minHeight: 48 }}
        placeholder={t('images_alt_placeholder')}
        defaultValue={item.alt_text ?? ''}
        onEndEditing={(e) => onAltChange(item.id, e.nativeEvent.text)}
      />
      <Pressable
        onPressIn={drag}
        accessibilityLabel="reorder"
        style={{ minWidth: 48, minHeight: 48, justifyContent: 'center', alignItems: 'center' }}
      >
        <Text style={{ fontSize: 20 }}>⋮⋮</Text>
      </Pressable>
      <Pressable
        onPress={() => setConfirmDeleteId(item.id)}
        accessibilityLabel={t('images_delete_confirm')}
        style={{ minWidth: 48, minHeight: 48, justifyContent: 'center', alignItems: 'center', backgroundColor: '#dc2626', borderRadius: 4 }}
      >
        <Text style={{ color: 'white', fontSize: 14 }}>{t('images_delete_yes')}</Text>
      </Pressable>
    </View>
  );

  return (
    <View style={{ flex: 1, padding: 16 }}>
      <View style={{ flexDirection: 'row', justifyContent: 'space-between', marginBottom: 12 }}>
        <Text style={{ fontSize: 20, fontWeight: '600' }}>{t('images_title')}</Text>
        <Text>{t('images_count', { n: images.length })}</Text>
      </View>
      <Pressable
        onPress={onAdd}
        disabled={uploading || images.length >= 10}
        style={{
          backgroundColor: images.length >= 10 ? '#a3a3a3' : '#15803d',
          padding: 14, borderRadius: 4, marginBottom: 16, minHeight: 48,
          alignItems: 'center', justifyContent: 'center',
        }}
      >
        {uploading
          ? <ActivityIndicator color="white" />
          : <Text style={{ color: 'white', fontSize: 16 }}>+ {t('images_add')}</Text>}
      </Pressable>

      <DraggableFlatList
        data={images}
        keyExtractor={(item) => item.id}
        renderItem={renderItem}
        onDragEnd={({ data }) => void onReorder(data)}
      />

      <Modal visible={!!confirmDeleteId} transparent animationType="slide">
        <View style={{ flex: 1, justifyContent: 'flex-end', backgroundColor: 'rgba(0,0,0,0.4)' }}>
          <View style={{ backgroundColor: 'white', padding: 20, borderTopLeftRadius: 12, borderTopRightRadius: 12 }}>

            </Pressable>
          </View>
        </View>
```

```tsx
function altFor(image: PublicImageRow, productName: string, index: number): string {
  return image.alt_text ?? `${productName} – तस्वीर ${index + 1}`;
}

export function ProductGallery({ images, productName }: { images: PublicImageRow[]; productName: string }) {
  const [activeIndex, setActiveIndex] = useState(0);
  const [lightboxOpen, setLightboxOpen] = useState(false);
  const flatListRef = useRef<FlatList<PublicImageRow>>(null);

  if (images.length === 0) {
    return (
      <View style={{ width: screenWidth, height: screenWidth, backgroundColor: '#F5EDDD', justifyContent: 'center', alignItems: 'center' }}>
        <Text>तस्वीर उपलब्ध नहीं</Text>
      </View>
    );
  }

  const renderItem = ({ item, index }: { item: PublicImageRow; index: number }) => (
    <Pressable
      onPress={() => setLightboxOpen(true)}
      accessibilityRole="image"
      accessibilityLabel={altFor(item, productName, index)}
    >
      <Image
        source={{ uri: item.default_url }}
        placeholder={{ uri: item.placeholder_url }}
        style={{ width: screenWidth, height: screenWidth, backgroundColor: '#F5EDDD' }}
        contentFit="cover"
      />
    </Pressable>
  );

  return (
    <>
      <FlatList
        ref={flatListRef}
        data={images}
        renderItem={renderItem}
        keyExtractor={(item) => item.id}
        horizontal
        pagingEnabled
        showsHorizontalScrollIndicator={false}
        onMomentumScrollEnd={(e) => {
          setActiveIndex(Math.round(e.nativeEvent.contentOffset.x / screenWidth));
        }}
      />

      {/* Dot indicators */}
      <View style={{ flexDirection: 'row', justifyContent: 'center', marginTop: 8 }}>
        {images.map((_, i) => (
          <View
            key={i}
            style={{
              width: 8, height: 8, borderRadius: 4, marginHorizontal: 4,
              backgroundColor: i === activeIndex ? '#15803d' : '#d4d4d4',
            }}
          />
        ))}
      </View>

      <Modal visible={lightboxOpen} transparent animationType="fade" onRequestClose={() => setLightboxOpen(false)}>
        <Pressable
          onPress={() => setLightboxOpen(false)}
          style={{ flex: 1, backgroundColor: 'rgba(0,0,0,0.95)', justifyContent: 'center', alignItems: 'center' }}
          accessibilityLabel="Close"
        >
          {images[activeIndex] && (
            <Image
              source={{ uri: images[activeIndex]!.default_url }}
              style={{ width: '100%', height: '70%' }}
              contentFit="contain"
            />
          )}
        </Pressable>
      </Modal>
    </>
  );
}
```
